# plugin CMake file to be included in the VPX main project CMakeLists

# --- ScoreTracker plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(SCORETRACKER_PLUGIN_SOURCES
   plugins/scoretracker/PluginMain.cpp
   plugins/scoretracker/ScoreTracker.cpp
   plugins/scoretracker/B2STracker.cpp
   plugins/scoretracker/ScoresFileWriter.cpp
   third-party/include/mongoose/mongoose.c
)

if(BUILD_SHARED)
   add_library(ScoreTrackerPlugin MODULE ${SCORETRACKER_PLUGIN_SOURCES})

   target_include_directories(ScoreTrackerPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/scoretracker
   )

   if(PluginPlatform STREQUAL "windows" OR PluginPlatform STREQUAL "windows-mingw")
      if(MSVC)
         add_compile_options(
            $<$<CONFIG:RELEASE>:/Ob2>
            $<$<CONFIG:RELEASE>:/O2>
            $<$<CONFIG:RELEASE>:/Oi>
            $<$<CONFIG:RELEASE>:/arch:SSE2>
            $<$<CONFIG:RELEASE>:/fp:fast>
            $<$<CONFIG:RELEASE>:/fp:except->
            $<$<CONFIG:RELEASE>:/Ot>
            $<$<CONFIG:RELEASE>:/GF>
            $<$<CONFIG:RELEASE>:/GS->
            $<$<CONFIG:RELEASE>:/Gy>
            $<$<CONFIG:RELEASE>:/GR>
            $<$<CONFIG:RELEASE>:/Oy>
            $<$<CONFIG:RELEASE>:/GT>
            $<$<CONFIG:RELEASE>:/GL>
         )
         set_target_properties(ScoreTrackerPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
         )
         target_compile_options(ScoreTrackerPlugin PUBLIC
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(ScoreTrackerPlugin PUBLIC
            $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
            $<$<CONFIG:RELEASE>:/OPT:REF>
            $<$<CONFIG:RELEASE>:/OPT:ICF>
            $<$<CONFIG:RELEASE>:/LTCG>
         )
      endif()
      target_link_directories(ScoreTrackerPlugin PUBLIC ${CMAKE_SOURCE_DIR}/third-party/build-libs/${PluginPlatform}-${PluginArch} ${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch})
      set_target_properties(ScoreTrackerPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "scoretracker-plugin"
      )
      if (PluginArch STREQUAL "x64")
         target_link_libraries(ScoreTrackerPlugin pinmame64)
      else()
         target_link_libraries(ScoreTrackerPlugin pinmame)
      endif()
   else()
      set_target_properties(ScoreTrackerPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "scoretracker-plugin"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(ScoreTrackerPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      else()
         set_target_properties(ScoreTrackerPlugin PROPERTIES
            INSTALL_RPATH "\$ORIGIN/../.."
            BUILD_WITH_INSTALL_RPATH TRUE
         )
      endif()
      target_link_directories(ScoreTrackerPlugin PUBLIC
         ${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}
      )
      target_link_libraries(ScoreTrackerPlugin PUBLIC pinmame)
   endif()

   add_custom_command(TARGET ScoreTrackerPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/scoretracker/plugin.cfg" "${PLUGINS_DIR}/scoretracker/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:ScoreTrackerPlugin>" "${PLUGINS_DIR}/scoretracker"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:ScoreTrackerPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS AND PluginPlatform STREQUAL "windows-mingw")
      add_custom_command(TARGET ScoreTrackerPlugin POST_BUILD
         COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/scoretracker/libgcc_s_seh-1.dll"
         COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libstdc++-6.dll" "${PLUGINS_DIR}/scoretracker/libstdc++-6.dll"
      )
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(ScoreTrackerPlugin STATIC ${SCORETRACKER_PLUGIN_SOURCES})

   target_include_directories(ScoreTrackerPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/scoretracker
   )

   set_target_properties(ScoreTrackerPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "scoretracker-plugin"
   )
endif()
